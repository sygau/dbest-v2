/* /* --- Modern Sidebar Gap Handling (Deepseek's approach) --- */
/* Default: No gap for mobile/tablet/touch devices */
.sidebar-wrapper .sidebar-nav {
  margin-top: 0;
  height: 100%;
}

/* Desktop/laptop: Add gap only if device is wide AND has hover (mouse/trackpad) */
@media (min-width: 1200px) and (hover: hover) {
  .sidebar-wrapper .sidebar-nav {
    margin-top: 70px;
    height: calc(100% - 70px);
  }
}

/* Optional: Explicitly target tablets in landscape (safety net) */
@media (min-width: 768px) and (max-width: 1199px) and (orientation: landscape) {
  .sidebar-wrapper .sidebar-nav {
    margin-top: 0;
    height: 100%;
  }
} 

// ---------- Google Analytics page_view on Swup navigation ----------
(function () {
  // Sends a page_view event to GA4 (gtag) or UA (ga)
  let lastTrackedPath = window.location.pathname;
  function sendGaPageView() {
    const currentPath = window.location.pathname;
    if (currentPath === lastTrackedPath) {
      // Prevent duplicate page_view for the same path
      return;
    }
    lastTrackedPath = currentPath;
    if (typeof gtag === 'function') {
      // GA4 (default)
      gtag('event', 'page_view', {
        page_location: window.location.href,
        page_path: window.location.pathname,
        page_title: document.title
      });
    } 
  }

  // Keep trying to hook into Swup until it exists
  function attachSwupHook() {
    if (window.swup && window.swup.hooks) {
      window.swup.hooks.on('content:replace', sendGaPageView);
    } else {
      // Swup not yet available; retry shortly
      setTimeout(attachSwupHook, 200);
    }
  }
  attachSwupHook();
})();
